414  static PrettyTable fromCSV(const std::string &filename, std::size_t columnCount,
415  std::size_t precision = 1, bool withHeaders = true, char splitor = ',') {
416  std::ifstream ifs(filename);
417  if (!ifs.is_open()) {
418  THROW_EXCEPTION("the csv file is not found in 'fromCSV'");
419  }
420  std::vector<std::string> headers;
421  std::string strLine;
422  // organize the headers
423  if (!withHeaders) {
424  headers.resize(columnCount);
425  std::size_t columnIndex = 0;
426  for (auto &elem : headers) {
427  elem = "Column(" + std::to_string(columnIndex++) + ")";
428  }
429  } else {
430  std::getline(ifs, strLine);
431  if (strLine.empty()) {
432  return PrettyTable(std::vector<std::string>());
433  } else {
434  headers = PrettyTable::split(strLine, splitor);
435  for (int i = 0; i < columnCount - headers.size(); ++i) {
436  headers.push_back("No-Header");
437  }
438  }
439  }
440  // organize the data
441  PrettyTable tab(headers);
442 
443  while (std::getline(ifs, strLine)) {
444  auto items = PrettyTable::split(strLine, splitor);
445  for (int i = 0; i != std::min(items.size(), columnCount); ++i) {
446  tab._data.at(i).push_back(items.at(i));
447  }
448  for (int i = 0; i < columnCount - items.size(); ++i) {
449  tab._data.at(i + std::min(items.size(), columnCount)).push_back("No-Data");
450  }
451  }
452  ifs.close();
453  return tab;
454  }
